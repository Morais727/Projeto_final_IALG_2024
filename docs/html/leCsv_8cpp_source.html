324int partition(acomodacoes* registros, int low, int high, Comparador comparador)
325{
326 acomodacoes pivot = registros[high]; // Escolhe o último elemento como pivô
327 int i = (low - 1);
328
329 for (int j = low; j <= high - 1; j++)
330 {
331 if (comparador(registros[j], pivot))
332 { // Usa o critério de comparação
333 i++;
334 swap(registros[i], registros[j]);
335 }
336 }
337 swap(registros[i + 1], registros[high]);
338 return (i + 1);
339}

348void quickSort(acomodacoes* registros, int low, int high, Comparador comparador)
349{
350 if (low < high)
351 {
352 int pi = partition(registros, low, high, comparador);
353
354 quickSort(registros, low, pi - 1, comparador);
355 quickSort(registros, pi + 1, high, comparador);
356 }
357}
